#!/bin/bash
# Infrastructure assessment script - run on 192.168.168.31

echo "=== SYSTEM INFORMATION ==="

echo ""

echo "=== GPU ==="
nvidia-smi --query-gpu=index,name,memory.total,driver_version,compute_cap --format=csv
echo ""

echo "=== CUDA & cuDNN ==="
nvcc --version 2>/dev/null || echo "CUDA not found"
find /usr/local/cuda -name "libcudnn*" 2>/dev/null | head -3 || echo "cuDNN not found"
echo ""

echo "=== DOCKER ==="
docker --version
docker compose version 2>/dev/null || docker-compose --version 2>/dev/null
docker run --rm --runtime=nvidia nvidia/cuda:12.4-base nvidia-smi | head -5 || echo "GPU Docker access failed"
echo ""

echo "=== NETWORK ==="
ip addr | grep "inet " | grep -v 127.0.0.1
echo ""

echo "=== NAS MOUNTS ==="
mount | grep /mnt/nas
echo ""

echo "=== AVAILABLE SPACE ==="
du -sh / /mnt/nas-* 2>/dev/null | head -5
